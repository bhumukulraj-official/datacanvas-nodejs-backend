# Portfolio Website Backend Implementation Guide

This document provides guidance on implementing the portfolio website backend according to the specifications in the API documentation, architecture design, and database schema.

## Getting Started

### Module Implementation Order

Implement the modules in the following order to build a working foundation first:

1. Core Setup & Configuration
2. Authentication Module
3. Profile Module
4. Projects Module
5. Media Module
6. Blog Module
7. Contact & Notifications
8. Advanced Features

```javascript
      }
```

```javascript
// src/modules/auth/services/auth.service.js
const User = require('../models/user.model');
const { ValidationError, AuthenticationError } = require('../../../shared/errors');

exports.registerUser = async (userData) => {
  // Check if email already exists
  const existingUser = await User.findOne({ where: { email: userData.email } });
  if (existingUser) {
    throw new ValidationError('Email already exists', { code: 'AUTH_005' });
  }
  
  // Create new user
  const user = await User.create({
    email: userData.email,
    password_hash: userData.password,
    name: userData.name,
    role: userData.role || 'user'
  });
  
  return user;
};

exports.validateCredentials = async (email, password) => {
  // Find user by email
  const user = await User.findOne({ where: { email } });
  if (!user) {
    throw new AuthenticationError('Invalid credentials');
  }
  
  // Validate password
  const isValid = await user.validatePassword(password);
  if (!isValid) {
    throw new AuthenticationError('Invalid credentials');
  }
  
  return user;
};
```

```javascript
// src/modules/media/services/media.service.js
const Media = require('../models/media.model');
const { v4: uuidv4 } = require('uuid');
const fs = require('fs');
const path = require('path');
const { Storage } = require('@google-cloud/storage');
const { NotFoundError, ValidationError } = require('../../../shared/errors');

// Initialize storage
const storage = new Storage();
const bucket = storage.bucket(process.env.STORAGE_BUCKET);

exports.uploadMedia = async (file, userId, type, description = '') => {
  // Validate file type
  const validTypes = {
    image: ['image/jpeg', 'image/png', 'image/gif', 'image/webp'],
    document: ['application/pdf', 'application/msword', 'application/vnd.openxmlformats-officedocument.wordprocessingml.document'],
    video: ['video/mp4', 'video/webm']
  };
  
  if (!validTypes[type].includes(file.mimetype)) {
    throw new ValidationError('Invalid file type', { code: 'FILE_002' });
  }
  
  // Validate file size
  const maxSizes = {
    image: 5 * 1024 * 1024, // 5MB
    document: 10 * 1024 * 1024, // 10MB
    video: 50 * 1024 * 1024 // 50MB
  };
  
  if (file.size > maxSizes[type]) {
    throw new ValidationError('File too large', { code: 'FILE_001' });
  }
  
  // Generate unique filename
  const fileExtension = path.extname(file.originalname);
  const fileName = `${uuidv4()}${fileExtension}`;
  const filePath = `uploads/${type}s/${fileName}`;
  
  // Upload to storage bucket
  await bucket.upload(file.path, {
    destination: filePath,
    metadata: {
      contentType: file.mimetype
    }
  });
  
  // Create media record
  const media = await Media.create({
    user_id: userId,
    url: `https://storage.googleapis.com/${process.env.STORAGE_BUCKET}/${filePath}`,
    type,
    size: file.size,
    filename: file.originalname,
    description,
    metadata: {
      contentType: file.mimetype
    }
  });
  
  // Remove temporary file
  fs.unlinkSync(file.path);
  
  return media;
};
```

## 6. WebSocket Implementation

```javascript
// src/shared/websocket/server.js
const WebSocket = require('ws');
const http = require('http');
const url = require('url');
const { v4: uuidv4 } = require('uuid');
const { verifyAccessToken } = require('../../modules/auth/services/token.service');
const logger = require('../utils/logger');
const redis = require('../config/redis');

// WebSocket configuration
const WS_CONFIG = {
  HEARTBEAT_INTERVAL: 30000, // 30 seconds
  CONNECTION_TIMEOUT: 30000, // 30 seconds
  MAX_MESSAGE_SIZE: 1024 * 1024, // 1MB
  MAX_MESSAGES_PER_MINUTE: 100,
  MAX_CONNECTIONS_PER_IP: 10,
  RECONNECT_INTERVALS: [1000, 2000, 4000, 8000, 16000], // Exponential backoff
  MAX_RETRY_ATTEMPTS: 5
};

// Error codes matching API documentation
const ERROR_CODES = {
  CONNECTION_FAILED: 'WS_001',
  AUTHENTICATION_FAILED: 'WS_002',
  INVALID_MESSAGE_FORMAT: 'WS_003',
  RATE_LIMIT_EXCEEDED: 'WS_004',
  CONNECTION_TIMEOUT: 'WS_005',
  SERVER_ERROR: 'WS_006'
};

const setupWebSocketServer = (server) => {
  const wss = new WebSocket.Server({ 
    noServer: true,
    maxPayload: WS_CONFIG.MAX_MESSAGE_SIZE
  });
  
  // Track connections by IP
  const connections = {};
  
  // Rate limiting for messages
  const messageCounts = {};
  
  // Heartbeat to keep connections alive
  const heartbeat = () => {
    wss.clients.forEach((ws) => {
      if (ws.isAlive === false) {
        // Connection timeout
        sendErrorMessage(ws, ERROR_CODES.CONNECTION_TIMEOUT, 'Connection timeout', false);
        return ws.terminate();
      }
      
      ws.isAlive = false;
      ws.ping();
      
      // Send status message as heartbeat
      sendStatusMessage(ws, 'connected', 'Connection alive', 200);
    });
  };
  
  const heartbeatInterval = setInterval(heartbeat, WS_CONFIG.HEARTBEAT_INTERVAL);
  
  // Handle connections
  wss.on('connection', (ws, request, user) => {
    // Store user ID with connection
    ws.userId = user.sub;
    ws.connectionId = uuidv4();
    ws.isAlive = true;
    
    // Store connection in database for tracking
    storeConnection(ws.connectionId, user.sub, request.connection.remoteAddress, request.headers['user-agent']);
    
    logger.info(`WebSocket connection established for user ${user.sub}`, {
      connectionId: ws.connectionId,
      userId: user.sub
    });
    
    // Send welcome message
    sendStatusMessage(ws, 'connected', 'Connection established', 200);
    
    // Handle pong response (heartbeat)
    ws.on('pong', () => {
      ws.isAlive = true;
      updateConnectionHeartbeat(ws.connectionId);
    });
    
    // Handle messages
    ws.on('message', (message) => {
      try {
        // Check rate limiting
        if (isRateLimited(ws.userId)) {
          return sendErrorMessage(
            ws, 
            ERROR_CODES.RATE_LIMIT_EXCEEDED, 
            'Too many messages, please slow down', 
            true,
            60
          );
        }
        
        // Parse message
        const parsedMessage = JSON.parse(message);
        
        // Validate message format
        if (!validateMessageFormat(parsedMessage)) {
          return sendErrorMessage(ws, ERROR_CODES.INVALID_MESSAGE_FORMAT, 'Invalid message format', false);
        }
        
        // Process message based on type
        processMessage(ws, parsedMessage);
        
      } catch (error) {
        // Send error message for JSON parse errors
        logger.error(`WebSocket message error: ${error.message}`, {
          userId: ws.userId,
          connectionId: ws.connectionId,
          error: error.message
        });
        
        sendErrorMessage(ws, ERROR_CODES.INVALID_MESSAGE_FORMAT, 'Invalid message format', false);
      }
    });
    
    // Handle connection close
    ws.on('close', () => {
      logger.info(`WebSocket connection closed for user ${user.sub}`, {
        connectionId: ws.connectionId,
        userId: ws.userId
      });
      
      // Update connection record
      updateConnectionClosed(ws.connectionId);
      
      // Clean up rate limit counters
      delete messageCounts[ws.userId];
      
      // Remove from IP tracking
      const ip = request.connection.remoteAddress;
      if (connections[ip]) {
        connections[ip]--;
        if (connections[ip] <= 0) {
          delete connections[ip];
        }
      }
    });
    
    // Handle errors
    ws.on('error', (error) => {
      logger.error(`WebSocket error: ${error.message}`, {
        userId: ws.userId,
        connectionId: ws.connectionId,
        error: error.message
      });
    });
  });
  
  // Handle upgrade requests
  server.on('upgrade', async (request, socket, head) => {
    const { pathname, query } = url.parse(request.url, true);
    
    if (pathname === '/api/v1/ws/notifications') {
      try {
        // Check connection limit by IP
        const ip = request.connection.remoteAddress;
        if (!connections[ip]) {
          connections[ip] = 0;
        }
        
        if (connections[ip] >= WS_CONFIG.MAX_CONNECTIONS_PER_IP) {
          socket.write('HTTP/1.1 429 Too Many Connections\r\n\r\n');
          socket.destroy();
          return;
        }
        
        // Update connection count
        connections[ip]++;
        
        // Verify token
        const token = query.token;
        if (!token) {
          socket.write('HTTP/1.1 401 Unauthorized\r\n\r\n');
          socket.destroy();
          return;
        }
        
        const user = verifyAccessToken(token);
        
        // Authenticate the connection
        wss.handleUpgrade(request, socket, head, (ws) => {
          wss.emit('connection', ws, request, user);
        });
    } catch (error) {
        logger.error(`WebSocket authentication failed: ${error.message}`, {
          ip: request.connection.remoteAddress,
          path: pathname
        });
        
        socket.write('HTTP/1.1 401 Unauthorized\r\n\r\n');
        socket.destroy();
      }
    } else {
      socket.write('HTTP/1.1 404 Not Found\r\n\r\n');
      socket.destroy();
    }
  });
  
  // Clean up on server close
  wss.on('close', () => {
    clearInterval(heartbeatInterval);
  });
  
  return wss;
};

/**
 * Send formatted status message
 * @param {WebSocket} ws WebSocket connection
 * @param {string} status Status type
 * @param {string} message Status message
 * @param {number} code Status code
 */
function sendStatusMessage(ws, status, message, code) {
      ws.send(JSON.stringify({
    type: 'status',
    payload: {
      status,
      message,
      code
    },
        timestamp: new Date().toISOString(),
    messageId: uuidv4()
      }));
    }

/**
 * Send formatted error message
 * @param {WebSocket} ws WebSocket connection
 * @param {string} code Error code
 * @param {string} message Error message
 * @param {boolean} reconnect Whether client should reconnect
 * @param {number} retryAfter Seconds to wait before retry
 */
function sendErrorMessage(ws, code, message, reconnect, retryAfter = 0) {
  ws.send(JSON.stringify({
    type: 'error',
    payload: {
      code,
      message,
      reconnect,
      retryAfter
    },
    timestamp: new Date().toISOString(),
    messageId: uuidv4()
  }));
}

 * Send notification message

}
```

## 7. Rate Limiting Implementation

```javascript
// src/shared/middleware/rate-limit.middleware.js
const redis = require('../config/redis');
const { RateLimitError } = require('../errors');

const rateLimit = (options) => {
  const {
    windowMs = 60 * 1000, // 1 minute
    max = 100, // 100 requests per windowMs
    message = 'Too many requests, please try again later',
    standardHeaders = true, // Send X-RateLimit-* headers
    keyGenerator = (req) => req.ip // IP-based rate limiting
  } = options;
  
  return async (req, res, next) => {
    try {
      const key = `rate-limit:${keyGenerator(req)}`;
      
      // Get current count from Redis
      const currentCount = await redis.get(key);
      
      if (!currentCount) {
        // First request in window, set count to 1 with expiry
        await redis.set(key, 1, 'EX', windowMs / 1000);
        
        if (standardHeaders) {
          res.setHeader('X-RateLimit-Limit', max);
          res.setHeader('X-RateLimit-Remaining', max - 1);
          res.setHeader('X-RateLimit-Reset', Math.floor(Date.now() / 1000) + (windowMs / 1000));
        }
        
        return next();
      }
      
      const count = parseInt(currentCount, 10);
      
      if (count >= max) {
        // Rate limit exceeded
        if (standardHeaders) {
          res.setHeader('X-RateLimit-Limit', max);
          res.setHeader('X-RateLimit-Remaining', 0);
          res.setHeader('X-RateLimit-Reset', Math.floor(Date.now() / 1000) + (windowMs / 1000));
          
          // Get TTL of the key
          const ttl = await redis.ttl(key);
          if (ttl > 0) {
            res.setHeader('Retry-After', ttl);
          }
        }
        
        throw new RateLimitError(message);
      }
      
      // Increment count
      await redis.incr(key);
      
      if (standardHeaders) {
        res.setHeader('X-RateLimit-Limit', max);
        res.setHeader('X-RateLimit-Remaining', max - count - 1);
        res.setHeader('X-RateLimit-Reset', Math.floor(Date.now() / 1000) + (windowMs / 1000));
      }
      
      next();
    } catch (error) {
      if (error instanceof RateLimitError) {
        return res.status(429).json({
          success: false,
          error: {
            code: 'RATE_001',
            message: error.message,
            timestamp: new Date().toISOString()
          }
        });
      }
      next(error);
    }
  };
};

// Define different rate limiters
const publicApiLimiter = rateLimit({
  windowMs: 60 * 1000, // 1 minute
  max: 100 // 100 requests per minute
});

const adminApiLimiter = rateLimit({
  windowMs: 60 * 1000, // 1 minute
  max: 1000 // 1000 requests per minute
});

const websocketLimiter = rateLimit({
  windowMs: 60 * 1000, // 1 minute
  max: 10 // 10 connections per minute
});

module.exports = {
  publicApiLimiter,
  adminApiLimiter,
  websocketLimiter
};
```

```javascript
// src/shared/middleware/auth.middleware.js
const { verifyAccessToken } = require('../../modules/auth/services/token.service');
const User = require('../../modules/auth/models/user.model');
const { AuthenticationError, TokenExpiredError, InvalidTokenError, PermissionError } = require('../errors');

exports.requireAuth = async (req, res, next) => {
  try {
    // Get the token from the Authorization header
    const authHeader = req.headers.authorization;
    if (!authHeader || !authHeader.startsWith('Bearer ')) {
      throw new AuthenticationError('Missing or invalid authorization header');
    }
    
    // Extract the token
    const token = authHeader.split(' ')[1];
    if (!token) {
      throw new AuthenticationError('Missing token');
    }
    
    // Verify the token
    const payload = verifyAccessToken(token);
    
    // Get user from database
    const user = await User.findByPk(payload.sub);
    if (!user) {
      throw new InvalidTokenError('User not found');
    }
    
    // Attach user to request
    req.user = user;
    
    next();
  } catch (error) {
    if (error instanceof TokenExpiredError || error instanceof InvalidTokenError) {
      return next(error);
    }
    next(new AuthenticationError(error.message));
  }
};

exports.requireAdmin = [
  exports.requireAuth,
  async (req, res, next) => {
    if (req.user.role !== 'admin') {
      return next(new PermissionError('Admin access required'));
    }
    next();
  }
];
```
